<script>
const canvas = document.getElementById("game");
const ctx = canvas.getContext("2d");

const box = 20;
let snake = [{ x: 9 * box, y: 9 * box }];
let direction = "RIGHT";

let food = {
  x: Math.floor(Math.random() * 18) * box,
  y: Math.floor(Math.random() * 18) * box
};

function setDir(dir) {
  if (dir === "LEFT" && direction !== "RIGHT") direction = dir;
  if (dir === "RIGHT" && direction !== "LEFT") direction = dir;
  if (dir === "UP" && direction !== "DOWN") direction = dir;
  if (dir === "DOWN" && direction !== "UP") direction = dir;
}

// swipe support
let startX = 0, startY = 0;
canvas.addEventListener("touchstart", e => {
  startX = e.touches[0].clientX;
  startY = e.touches[0].clientY;
});
canvas.addEventListener("touchend", e => {
  let dx = e.changedTouches[0].clientX - startX;
  let dy = e.changedTouches[0].clientY - startY;
  if (Math.abs(dx) > Math.abs(dy)) {
    if (dx > 0) setDir("RIGHT");
    else setDir("LEFT");
  } else {
    if (dy > 0) setDir("DOWN");
    else setDir("UP");
  }
});

function bodyCollision(head, body) {
  for (let i = 1; i < body.length; i++) {
    if (head.x === body[i].x && head.y === body[i].y) return true;
  }
  return false;
}

function draw() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  snake.forEach((s, i) => {
    ctx.fillStyle = i === 0 ? "lime" : "green";
    ctx.fillRect(s.x, s.y, box, box);
  });

  ctx.fillStyle = "red";
  ctx.fillRect(food.x, food.y, box, box);

  let headX = snake[0].x;
  let headY = snake[0].y;

  if (direction === "LEFT") headX -= box;
  if (direction === "RIGHT") headX += box;
  if (direction === "UP") headY -= box;
  if (direction === "DOWN") headY += box;

  // wall wrap
  if (headX < 0) headX = canvas.width - box;
  if (headY < 0) headY = canvas.height - box;
  if (headX >= canvas.width) headX = 0;
  if (headY >= canvas.height) headY = 0;

  if (headX === food.x && headY === food.y) {
    food = {
      x: Math.floor(Math.random() * 18) * box,
      y: Math.floor(Math.random() * 18) * box
    };
  } else {
    snake.pop();
  }

  const newHead = { x: headX, y: headY };

  if (bodyCollision(newHead, snake)) {
    clearInterval(game);
    alert("Game Over üêç");
  }

  snake.unshift(newHead);
}

let game = setInterval(draw, 140);
</script>
